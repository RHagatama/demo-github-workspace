# General command on linux terminal

## time and date
cal
date

## device usage or file size
df -h
du -sh *

## exploring
pwd
ls -lah
cd ~
which python3

## viewing files
less hello.sh
cat hello.sh

## create or modifying file
touch newfile.txt
mkdir newdir
mkdir -p moredir/dir1/dir2
rm -rf moredir
mv newfile.txt newdir
chmod +x filename

ps
./sleeper.sh
jobs
fg
